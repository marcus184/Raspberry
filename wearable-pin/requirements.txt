# Python dependencies for Wearable Pin Raspberry Pi Project
#
# This project includes camera capture, microphone recording, button control, and LED control
# for Raspberry Pi Zero 2W wearable applications.
#
# ============================================================================
# INSTALLATION INSTRUCTIONS
# ============================================================================
#
# On Raspberry Pi, install system packages first:
#   sudo apt-get update
#   sudo apt-get install -y python3-pip python3-gpiozero libcamera-apps alsa-utils
#
# Then install Python packages via pip:
#   sudo pip3 install -r requirements.txt
#
# Note: Some packages may require sudo for hardware access
#
# ============================================================================
# PYTHON PACKAGES (install via pip)
# ============================================================================

# GPIO control for buttons and sensors
gpiozero>=1.6.2

# WS2812/NeoPixel LED control (CircuitPython)
adafruit-circuitpython-neopixel>=1.10.0

# HTTP requests for cloud upload
requests>=2.28.0

# Camera library (optional - current code uses rpicam-still command-line tool)
# picamera2>=0.3.12
# Note: picamera2 is typically installed via apt-get on Raspberry Pi OS:
#   sudo apt-get install -y python3-picamera2
# If installing via pip (not recommended), uncomment the line above

# ============================================================================
# SYSTEM PACKAGES (install via apt-get)
# ============================================================================
#
# Required system packages (not installable via pip):
#   - python3-gpiozero      # GPIO library (can also use pip version)
#   - libcamera-apps        # Camera command-line tools (rpicam-still)
#   - alsa-utils            # Audio recording tools (arecord)
#
# Install with:
#   sudo apt-get install -y python3-gpiozero libcamera-apps alsa-utils
#
# ============================================================================
# STANDARD LIBRARY MODULES (no installation needed)
# ============================================================================
#
# These are built into Python and don't require installation:
#   - os
#   - sys
#   - time
#   - datetime
#   - pathlib
#   - subprocess
#   - platform
#
# ============================================================================
# PROJECT STRUCTURE
# ============================================================================
#
# Main scripts:
#   - button_capture.py    # Button-triggered camera capture (uses rpicam-still)
#   - mic_test.py          # Button-triggered microphone recording (uses arecord)
#   - led_test.py          # WS2812/NeoPixel LED control (uses CircuitPython neopixel)
#   - cloud_upload_test.py # Cloud file upload to remote server (uses requests)
#   - check_environment.py # Environment validation script
#
# ============================================================================
# HARDWARE REQUIREMENTS
# ============================================================================
#
# - Raspberry Pi Zero 2W (or compatible)
# - Arducam 5MP OV5647 camera (or compatible)
# - PH0645 I2S microphone
# - WS2812/NeoPixel LED (optional)
# - Momentary push buttons (GPIO 4 for camera, GPIO 23 for microphone)
#
# ============================================================================
